237/* Open SERIAL channel*/



238static int SERIAL_open(const char *port)



239{



240    int fd;



241    int retries = 0;



242    /* O_NDELAY: disregard DCD signal line state



243     * O_NOCTTY: we don't want to be the controlling terminal */



244    fflush(stdout);



245    while (retries < 15)



246    {



247        fd = open(port, O_RDWR | O_NOCTTY | O_NONBLOCK);



248        if (fd != -1) 



249        {



250            break;



251        }



252        // sleep a bit and retry. There seems to be a NuttX bug



253        // that can cause ttyACM0 to not be available immediately,



254        // but a small delay can fix it



255        usleep(50000);



256        retries++;



257    }



258    if (fd == -1)



259    {



260        perror("SERIAL_open/open");



261    }



262    



263    /*flush both data received but not read, and data written but not transmitted.*/



264    tcflush(fd, TCIOFLUSH);



265    return fd;



266}

268/* Return device ID given serial port name */



269int getCurrSciDev(const char *name)



270{



271    int i;



272    



273    for (i = NUM_MAX_SCI_DEV - 1; i > -1; i--) {



274        if ((name != NULL) && (sciDev[i].portname != NULL) &&



275                (strncmp(name, sciDev[i].portname, MAX_DEV_NAME) == 0)) {



276            break;



277        }



278    }



279    



280    return i;



281}



282/* Allocate device*/



283int Serial_alloc(const char *name)



284{



285    int i;



286    



287    for (i = 0; i < NUM_MAX_SCI_DEV; i++)



288    {



289        if (sciDev[i].portname == NULL)



290        {



291            break;



292        }



293    }



294    if (i >= NUM_MAX_SCI_DEV)



295    {



296        fprintf(stderr, "Cannot allocate a new device for %s: [%d]\n",



297                name, i);



298        return -1;



299    }



300    sciDev[i].portname = strndup(name, MAX_DEV_NAME);



301    



302    return i;



303}

305/* Initialize a SCI */



306MW_Handle_Type MW_SCI_Open(void * SCIModule,



307        uint8_T isString,



308        uint32_T RxPin, /* Not used */



309        uint32_T TxPin)/*Not Used*/



310{



311    MW_Handle_Type SCIHandle = (MW_Handle_Type)NULL;



312    char * port;



313    int currSciDev;



314    



315    /* Check parameters */



316    if (0 == isString)



317    {



318        fprintf(stderr,"Only string as SCI Module name is supported.\n");



319        exit(-1);



320    }



321    else



322    {



323        /* Initialize the SCI Module*/



324        port = (char*)SCIModule;



325#if DEBUG



326        printf("In MW_SCI_Open. Port = %s\n", port);



327#endif



328        currSciDev = getCurrSciDev(port);



329        fprintf(stdout, "INIT: sciDevNo = %d\n", currSciDev);



330        if (currSciDev == -1)



331        {



332            currSciDev = Serial_alloc(port);



333            fprintf(stdout, "ALLOC: devNo = %d\n", currSciDev);



334            if (currSciDev == -1)



335            {



336                fprintf(stderr,"Error opening Serial bus (SERIAL_init/Alloc).\n");



337                exit(-1);



338            }



339        }



340        



341        if (sciDev[currSciDev].fd < 0)



342        {



343            sciDev[currSciDev].fd = SERIAL_open(port);



344            if (sciDev[currSciDev].fd < 0)



345            {



346                fprintf(stderr,"Error opening Serial bus (SERIAL_init/Open).\n");



347                exit(-1);



348            }



349        }



350        SCIHandle = (MW_Handle_Type)&sciDev[currSciDev];



351    }



352    return SCIHandle;



353}

355/* Set the SCI bus speed */



356MW_SCI_Status_Type MW_SCI_SetBaudrate(MW_Handle_Type SCIModuleHandle, uint32_T Baudrate)



357{



358    SCI_dev_t *sci;



359    struct termios options;



360    speed_t optBaud;



361    



362    if (NULL != (void *)SCIModuleHandle)



363    {



364        sci = (SCI_dev_t *)SCIModuleHandle;



365        



366        



367        if (strcmp(sci->portname,"/dev/ttyACM0") == 0)



368        {



369            Baudrate = AllBaudrates[MW_TTYACM0_BAUDRATE];



370#if DEBUG



371            PX4_INFO("Inside MW_SCI_SetBaudrate strcmp. Portname = %s.Baudrate = %u\n.",sci->portname,Baudrate);



372#endif



373        }



374        else if (strcmp(sci->portname,"/dev/ttyS0") == 0)



375        {



376            Baudrate = AllBaudrates[MW_TTYS0_BAUDRATE];



377        }



378        else if (strcmp(sci->portname,"/dev/ttyS1") == 0)



379        {



380            Baudrate = AllBaudrates[MW_TTYS1_BAUDRATE];



381        }



382        else if (strcmp(sci->portname,"/dev/ttyS2") == 0)



383        {



384            Baudrate = AllBaudrates[MW_TTYS2_BAUDRATE];



385#if DEBUG



386            PX4_INFO("Inside MW_SCI_SetBaudrate strcmp. Portname = %s.Baudrate = %u\n.",sci->portname,Baudrate);



387#endif



388        }



389        else if (strcmp(sci->portname,"/dev/ttyS3") == 0)



390        {



391            Baudrate = AllBaudrates[MW_TTYS3_BAUDRATE];



392        }



393        else if (strcmp(sci->portname,"/dev/ttyS4") == 0)



394        {



395            Baudrate = AllBaudrates[MW_TTYS4_BAUDRATE];



396        }



397        else if (strcmp(sci->portname,"/dev/ttyS5") == 0)



398        {



399            Baudrate = AllBaudrates[MW_TTYS5_BAUDRATE];



400        }



401        else if (strcmp(sci->portname,"/dev/ttyS6") == 0)



402        {



403            Baudrate = AllBaudrates[MW_TTYS6_BAUDRATE];



404#if DEBUG



405            PX4_INFO("Inside MW_SCI_SetBaudrate strcmp. Portname = %s.Baudrate = %u\n.",sci->portname,Baudrate);



406#endif



407        }



408        else



409        {



410            perror("SERIAL_SetBaudrate");



411            return MW_SCI_BUS_ERROR;



412        }



413        



414#if DEBUG



415        printf("Inside MW_SCI_SetBaudrate. Portname = %s. Baudrate = %u\n.",sci->portname,Baudrate);



416#endif



417        /* Set parameters of the serial connection*/



418//         optBaud = Baudrate;



419        switch (Baudrate)



420        {



421            case      50: optBaud =       B50; break;



422            case      75: optBaud =       B75; break;



423            case     110: optBaud =      B110; break;



424            case     134: optBaud =      B134; break;



425            case     150: optBaud =      B150; break;



426            case     200: optBaud =      B200; break;



427            case     300: optBaud =      B300; break;



428            case     600: optBaud =      B600; break;



429            case    1200: optBaud =     B1200; break;



430            case    1800: optBaud =     B1800; break;



431            case    2400: optBaud =     B2400; break;



432            case    4800: optBaud =     B4800; break;



433            case    9600: optBaud =     B9600; break;



434            case   19200: optBaud =    B19200; break;



435            case   38400: optBaud =    B38400; break;



436            case   57600: optBaud =    B57600; break;



437            case  115200: optBaud =   B115200; break;



438            case  230400: optBaud =   B230400; break;



439            case 1500000: optBaud =  B1500000; break;



440            case 2000000: optBaud =  B2000000; break;



441            case 2500000: optBaud =  B2500000; break;



442            case 3000000: optBaud =  B3000000; break;



443            default:



444                perror("SERIAL_SetBaudrate");



445                return MW_SCI_BUS_ERROR;



446        }



447        



448        if (tcgetattr(sci->fd, &options) < 0)



449        {



450            perror("SERIAL_GetAttribute");



451            return MW_SCI_BUS_ERROR;



452        }



453               



454        /* Enable the receiver and set local mode*/



455        options.c_cflag |= (CLOCAL     /*To not become port 'owner'*/



456                | CREAD);  /*Allow reading of incoming data*/



457        options.c_iflag &= ~(IXON /*Disable Software Flow Control*/



458                          | IXOFF | IXANY);



459        



460        options.c_lflag &= ~(ICANON   /*To have raw output*/



461                          | ECHO   /*Disable input character echo*/



462                          | ECHOE  /*Disable echo character erase*/



463                          | ISIG); /*Disable SIGINTR, SIGSUSP, SIGDSUSP,



464                                    * and SIGQUIT signals*/



465                



466        /* Set character read options*/



467        options.c_cc[VMIN]  = 0;



468        options.c_cc[VTIME] = 0;  /*This is a completely non-blocking read -*/



469        /* Local options. Configure for RAW input*/



470        



471        options.c_oflag &= ~OPOST; /*Disable Post processing of output*/



472        



473        /* Set baud rate*/



474        if (cfsetispeed(&options, optBaud) < 0)



475        {



476            perror("SERIAL_Set_ip_Speed");



477            return MW_SCI_BUS_ERROR;



478        }



479        if (cfsetospeed(&options, optBaud) < 0)



480        {



481            perror("SERIAL_Set_op_Speed");



482            return MW_SCI_BUS_ERROR;



483        }



484        



485        /* Set attributes*/



486        if (tcsetattr(sci->fd, TCSANOW, &options) < 0)



487        {



488            perror("SERIAL_SetAttribute");



489            return MW_SCI_BUS_ERROR;



490        }



491        else



492        {



493            printf("Serial Port Opened. fd = %d. BaudRate = %u\n", sci->fd,optBaud);



494        }



495        /*flush both data received but not read, and data written but not transmitted.*/



496        tcflush(sci->fd, TCIOFLUSH);



497    }



498    



499    return MW_SCI_SUCCESS;



500}

502/* Set SCI frame format */



503MW_SCI_Status_Type MW_SCI_SetFrameFormat(MW_Handle_Type SCIModuleHandle, uint8_T DataBitsLength, MW_SCI_Parity_Type Parity, MW_SCI_StopBits_Type StopBits)



504{



505    SCI_dev_t *sci;



506    struct termios options;



507    



508    if (NULL != (void *)SCIModuleHandle)



509    {



510        



511        sci = (SCI_dev_t *)SCIModuleHandle;



512        if (tcgetattr(sci->fd, &options) < 0)



513        {



514            perror("SERIAL_GetAttribute");



515            return MW_SCI_BUS_ERROR;



516        }



517        



518        /* Set data bits*/



519        options.c_cflag &= ~CSIZE;



520        



521        switch (DataBitsLength)



522        {



523            case 5:



524                options.c_cflag |= CS5;



525                break;



526            case 6:



527                options.c_cflag |= CS6;



528                break;



529            case 7:



530                options.c_cflag |= CS7;



531                break;



532            case 8:



533                options.c_cflag |= CS8;



534                break;



535            default:



536                perror("SERIAL_SetFrameFormat/DataBitsLength");



537                return MW_SCI_BUS_ERROR;



538        }



539               



540        if (strcmp(sci->portname,"/dev/ttyACM0") == 0)



541        {



542            Parity = MW_TTYACM0_PARITY;



543            StopBits = (2 * MW_TTYACM0_STOPBIT) + 1;



544#if DEBUG



545            PX4_INFO("Inside MW_SCI_SetFrameFormat strcmp. Portname = %s.Parity = %u. StopBits = %u\n.",sci->portname,Parity,StopBits);



546#endif



547        }



548        else if (strcmp(sci->portname,"/dev/ttyS0") == 0)



549        {



550            Parity = MW_TTYS0_PARITY;



551            StopBits = (2 * MW_TTYS0_STOPBIT) + 1;



552        }



553        else if (strcmp(sci->portname,"/dev/ttyS1") == 0)



554        {



555            Parity = MW_TTYS1_PARITY;



556            StopBits = (2 * MW_TTYS1_STOPBIT) + 1;



557        }



558        else if (strcmp(sci->portname,"/dev/ttyS2") == 0)



559        {



560            Parity = MW_TTYS2_PARITY;



561            StopBits = (2 * MW_TTYS2_STOPBIT) + 1;



562#if DEBUG



563            PX4_INFO("Inside MW_SCI_SetFrameFormat strcmp. Portname = %s.Parity = %u. StopBits = %u\n.",sci->portname,Parity,StopBits);



564#endif



565        }



566        else if (strcmp(sci->portname,"/dev/ttyS3") == 0)



567        {



568            Parity = MW_TTYS3_PARITY;



569            StopBits = (2 * MW_TTYS3_STOPBIT) + 1;



570        }



571        else if (strcmp(sci->portname,"/dev/ttyS4") == 0)



572        {



573            Parity = MW_TTYS4_PARITY;



574            StopBits = (2 * MW_TTYS4_STOPBIT) + 1;



575        }



576        else if (strcmp(sci->portname,"/dev/ttyS5") == 0)



577        {



578            Parity = MW_TTYS5_PARITY;



579            StopBits = (2 * MW_TTYS5_STOPBIT) + 1;



580        }



581        else if (strcmp(sci->portname,"/dev/ttyS6") == 0)



582        {



583            Parity = MW_TTYS6_PARITY;



584            StopBits = (2 * MW_TTYS6_STOPBIT) + 1;



585#if DEBUG



586            PX4_INFO("Inside MW_SCI_SetFrameFormat strcmp. Portname = %s.Parity = %u. StopBits = %u\n.",sci->portname,Parity,StopBits);



587#endif



588        }



589        else



590        {



591            perror("SERIAL_SetFrameFormat/Parity");



592            return MW_SCI_BUS_ERROR;



593        }



594        



595#if DEBUG



596        printf("Inside MW_SCI_SetFrameFormat. Portname = %s. Parity = %u. StopBits = %u\n.",sci->portname,Parity,StopBits);



597#endif



598        /* Set parity*/



599        switch (Parity)



600        {



601            case MW_SCI_PARITY_NONE:



602                options.c_cflag &= ~PARENB;



603                break;



604            case MW_SCI_PARITY_EVEN:



605                options.c_cflag |= PARENB;



606                options.c_cflag &= ~PARODD;



607                break;



608            case MW_SCI_PARITY_ODD:



609                options.c_cflag |= PARENB;



610                options.c_cflag |= PARODD;



611                break;



612            default:



613                perror("SERIAL_SetFrameFormat/Parity");



614                return MW_SCI_BUS_ERROR;



615        }



616        



617        /* Set stop bits (1 or 2)*/



618        switch (StopBits)



619        {



620            case MW_SCI_STOPBITS_1:



621                options.c_cflag &= ~CSTOPB;



622                break;



623            case MW_SCI_STOPBITS_2:



624                options.c_cflag |= CSTOPB;



625                break;



626            default:



627                perror("SERIAL_SetFrameFormat/StopBits");



628                return MW_SCI_BUS_ERROR;



629        }



630               



631        /* Set attributes*/



632        if (tcsetattr(sci->fd, TCSANOW, &options) < 0)



633        {



634            perror("SERIAL_SetAttribute");



635            return MW_SCI_BUS_ERROR;



636        }



637        /*flush both data received but not read, and data written but not transmitted.*/



638        tcflush(sci->fd, TCIOFLUSH);



639    }



640    



641    return MW_SCI_SUCCESS;



642}

644MW_SCI_Status_Type MW_SCI_GetDataBytesAvailable(MW_Handle_Type SCIModuleHandle, _Bool blockingMode, void * size, int32_T timeout)



645{



646    int out;



647    SCI_dev_t *sci;



648    int* databytesAvailable = (int*)size;



649    int error_counter = 0;



650    if (NULL != (void *)SCIModuleHandle)



651    {



652        sci = (SCI_dev_t *)SCIModuleHandle;



653        if (blockingMode)



654        {



655            //Blocking mode



656            px4_pollfd_struct_t fds[] =



657            {



658                { .fd = sci->fd,   .events = POLLIN },



659                /* there could be more file descriptors here, in the form like:



660                 * { .fd = other_sub_fd,   .events = POLLIN },



661                 */



662            };



663#if DEBUG



664            printf("Inside MW_SCI_GetDataBytesAvailable. Before Poll \n");



665#endif



666            timeout = timeout * 1000;



667            int poll_ret = px4_poll(fds, 1, timeout);



668#if DEBUG



669            printf("Inside MW_SCI_GetDataBytesAvailable. After Poll \n");



670#endif



671            /* handle the poll result */



672            if (poll_ret == 0)



673            {



674                /* this means none of our providers is giving us data */



675                PX4_ERR("Got no data within the timeout specified");



676                return MW_SCI_DATA_NOT_AVAILABLE;



677            }



678            else if (poll_ret < 0)



679            {



680                /* this is seriously bad - should be an emergency */



681                if (error_counter < 10 || error_counter % 50 == 0)



682                {



683                    /* use a counter to prevent flooding (and slowing us down) */



684                    PX4_ERR("ERROR return value from poll(): %d", poll_ret);



685                    return MW_SCI_BUS_ERROR;



686                }



687            }



688            else



689            {



690                if (fds[0].revents & POLLIN)



691                {



692                    sci->busy = 1; /*Set the busy flag*/



693                    out = ioctl (sci->fd, FIONREAD, (int*)databytesAvailable);



694                    sci->busy = 0; /*Reset the busy flag*/



695                    if(out < 0)



696                    {



697                        perror("MW_SCI_GetDataBytesAvailable");



698                        return MW_SCI_BUS_ERROR;



699                    }



700                }



701            }



702        }//End of Blocking mode



703        else



704            //Non-Blocking mode



705        {



706#if DEBUG



707            printf("Inside MW_SCI_GetDataBytesAvailable. Non-blocking mode \n");



708#endif



709            sci->busy = 1; /*Set the busy flag*/



710            out = ioctl (sci->fd, FIONREAD, (int*)databytesAvailable);



711            sci->busy = 0; /*Reset the busy flag*/



712            if(out < 0)



713            {



714                perror("MW_SCI_GetDataBytesAvailable");



715                return MW_SCI_BUS_ERROR;



716            }



717        } //End of Non-Blocking mode



718//         ptr_size = (uint8_T*)&databytesAvailable;



719#if DEBUG



720        printf("Debug Print in MW_SCI_GetDataBytesAvailable: available %lu bytes : %u bytes.\n ", *databytesAvailable,*(char*)size);



721#endif



722    }



723    return MW_SCI_SUCCESS;



724}
